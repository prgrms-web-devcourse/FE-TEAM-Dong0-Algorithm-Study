// 첫 풀이 >> 4, 7, 8 테케 실패

// 위아래 이동 수 : 절반보다 가까운 부분으로 위 혹은 아래 count 증가
// 옆 이동 수
// 1. 앞에서부터 마지막 한번에
// 2. 앞부분 왕복 후 뒷부분
// 3. 뒷부분 왕복 후 앞부분

function solution(name) {
  let sum = 0;
  name.split("").forEach((char) => {
    let count = 0;

    if (char.charCodeAt() <= 77) {
      count = char.charCodeAt() - 65;
    } else {
      count = 91 - char.charCodeAt();
    }
    sum += count;
  });

  const stack = [name.length - 1];
  let oneway = 0;
  let frontFlagIndex = 0;
  let rearFlagIndex = 0;

  // 왕복 없을 때
  name.split("").forEach((char, i) => {
    if (char !== "A") {
      oneway = i;
    }
  });

  // 앞 왕복
  name
    .slice(0, Math.ceil((name.length - 1) / 2) + 1)
    .split("")
    .forEach((char, i) => {
      if (char !== "A") {
        frontFlagIndex = i;
      }
    });

  // 뒤 왕복
  name
    .slice(Math.ceil((name.length - 1) / 2) + 1)
    .split("")
    .reverse()
    .forEach((char, i) => {
      if (char !== "A") {
        rearFlagIndex = i + 1;
      }
    });

  return (
    sum +
    Math.min(
      oneway,
      frontFlagIndex * 2 + rearFlagIndex,
      rearFlagIndex * 2 + frontFlagIndex
    )
  );
}
